GUI libraries are all slow and complicated. Nimrod tries to be different. 

The idea is to implement a new library, that is portable by just implementing
a common low-level layer. 

Low level (platform-dependant)
------------------------------

proc newWindow(Title, x = -1, y = -1, width = 600, height = 400, 
               flags: TWindowFlags): PWindow

proc waitForEvent(): TEvent
proc pollEvent(): TEvent
proc sendEvent(e: TEvent)

proc sendToClipboard()
proc receiveFromClipboard()

proc putPixel(w: PWindow, x, y: int, color: TColor)
proc getPixel(w: PWindow, x, y: int): TColor

# ... other drawing primitives ...
